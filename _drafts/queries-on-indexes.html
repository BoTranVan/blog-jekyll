a. write these queries: 1...2..3..4..5..6..
b. how would you improve indexes for these queries?
solutions:

1. SELECT Country.name, City.name FROM Country JOIN City ON (Country.CODE = City.CountryCode);

2. SELECT Country.name, City.name FROM Country JOIN City ON (Country.Capital = City.ID);

3. SELECT Country.Name FROM Country JOIN CountryLanguage on (CountryLanguage.CountryCode = Country.Code) WHERE Language='English';

4. SELECT Country.name, City.name FROM Country JOIN City ON (Country.Capital = City.ID)
WHERE City.Population > 1000000;

5. SELECT Country.name, City.name FROM Country JOIN City ON (Country.CODE = City.CountryCode)
WHERE City.Population > 1000000;

6. SELECT Country.name, City.name FROM Country JOIN City ON (Country.Capital = City.ID)
WHERE City.Population > 1000000 AND Continent='Europe';
